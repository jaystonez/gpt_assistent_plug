// Add this function to your plugin file to add the admin menu item
add_action('admin_menu', array('ChatGPT_Assistant', 'add_features_menu'));

class ChatGPT_Assistant {
    // Function to add the admin menu item
    public static function add_features_menu() {
        add_menu_page(
            'ChatGPT Features',         // Page title
            'ChatGPT Features',         // Menu title
            'manage_options',           // Capability required to access
            'chatgpt-features',         // Menu slug
            array(__CLASS__, 'features_page') // Callback function to display content
        );
    }

    // Function to display the features on the admin page
    public static function features_page() {
        ?>
        <div class="wrap">
            <h1>ChatGPT Assistant Features</h1>
            <ul>
                <?php
                // List of all features (including hidden and VIP)
                $features = array(
                    'advanced' => 'Advanced Feature',
                    'developer' => 'Developer Tools',
                    'logs' => 'Logging',
                    'diagnostics' => 'Diagnostics',
                    'performance' => 'Performance Monitoring',
                    'beta' => 'Beta Features',
                    'vip' => 'VIP Features',
                    'debug' => 'Debug Mode',
                    'safe' => 'Safe Mode',
                    'api' => 'API Access',
                    'pro' => 'Pro Features',
                    'config' => 'Configuration Tools',
                    'compatibility' => 'Compatibility Checks',
                    'maintenance' => 'Maintenance Tools',
                    'secret' => 'Hidden Features',
                    'cleanup' => 'Cleanup Utilities',
                    'security' => 'Security Features',
                    'usage' => 'Usage Statistics',
                    'remote' => 'Remote Access',
                    'updates' => 'Update Management',
                    'hidden' => 'Hidden Features'
                );

                // Display each feature
                foreach ($features as $key => $feature) {
                    echo '<li><strong>' . esc_html($feature) . '</strong> (' . esc_html($key) . ')</li>';
                }
                ?>
            </ul>
        </div>
        <?php
    }
}
